# cancom_ipam_supernet (Resource)

  instance_id   = "<id of the managed instance this supernet shall belong to>"
  name_tag      = "<name>"
  supernet_cidr = "<the prefix in cidr notation>"
  description   = "<a description>"

  name_tag      = "supernet-from-tf"

}
